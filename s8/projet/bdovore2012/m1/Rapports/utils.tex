% Place un texte dans une fbox avec retour à la ligne automatique
% \fboxln{anything}
\newcommand{\fboxln}[1]{%
  \fbox{%
    \begin{minipage}{0.9\textwidth}%
    #1%
    \end{minipage}%
  }
}
